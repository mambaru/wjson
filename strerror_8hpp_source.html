56class strerror

58public:

64 static std::string message( const json_error& e )
65 {
66 std::stringstream ss;
67 ss << strerror::what(e);
68 if ( e.expected_of() != fas_nullptr )
69 ss << " '" << e.expected_of() << "'";
70 return ss.str();
71 }
72
73 template<typename P>
74 static size_t where( const json_error& e, P beg, P end)
75 {
76 if ( !e ) return 0ul;
77
78 if ( std::distance(beg, end) < e.tail_of() )
79 return 0ul;
80
81 return static_cast<size_t>( std::distance(beg, end) - e.tail_of() );
82 }
83
84 template<typename P>
85 static std::string trace( const json_error& e, P beg, P end )
86 {
87 if ( !e )
88 return std::string(beg, end);
89
90 std::stringstream ss;
91 ss << std::string(beg, end - e.tail_of() ) << ( e.tail_of() ? ">>>" : "<<<") << std::string(end - e.tail_of(), end);
92 return ss.str();
93 }
94
95 template<typename P>
96 static std::string message_trace( const json_error& e, P beg, P end )
97 {
98 std::stringstream ss;
99 ss << strerror::what(e);
100 if ( e.expected_of() != fas_nullptr )
101 ss << " '" << e.expected_of() << "'";
102 if ( e )
103 {
104 ss << ": "
105 << std::string(beg, end - e.tail_of() )
106 << ( e.tail_of() ? ">>>" : "<<<")
107 << std::string(end - e.tail_of(), end);
108 }
109 return ss.str();
110 }
111
112private:
113
114 template<typename L, typename R>
115 static const char* what_( int code, fas::type_list<L, R> )
116 {
117 if ( code == static_cast<int>(L::value) )
118 return L()();
119 return strerror::what_(code, R());
120 }
121
122 template<typename L>
123 static const char* what_( int , fas::type_list<L, fas::empty_list> )
124 {
125 return L()();
126 }
